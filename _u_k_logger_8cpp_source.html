41 void UKLogger::log(const std::string& severity, const std::string& kind, const std::string& func,
42  const std::string& message, int line) {
43  // get current time
44  auto now = std::chrono::system_clock::now();
45 
46  // get number of milliseconds for the current second
47  // (remainder after division into seconds)
48  int millisToSeconds = 1000;
49  auto milliseconds = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()) % millisToSeconds;
50 
51  // convert to std::time_t in order to convert to std::tm (broken time)
52  auto timer = std::chrono::system_clock::to_time_t(now);
53 
54  std::tm tmStruct = {0};
55 #if (defined(_MSC_VER))
56  localtime_s(&tmStruct, &timer);
57 #else
58  localtime_r(&timer, &tmStruct); // POSIX
59 #endif
60  std::ostringstream stream;
61  stream << std::put_time(&tmStruct, "%Y.%m.%d %H:%M:%S")
62  << '.' << std::setfill('0') << std::setw(3) << milliseconds.count();
63  std::string timeStr = stream.str();
64  auto threadID = std::this_thread::get_id();
65  std::string function = func;
66  // remove leading const
67  std::size_t lenOfConst = std::string("const ").length();
68  if (0 == function.find("const ")) {
69  function.erase(0, lenOfConst);
70  }
71  // remove trailing const
72  if (function.size() - 6 == function.rfind(" const")) {
73  function.erase(function.size() - lenOfConst, lenOfConst);
74  }
75  // remove leading virtual
76  std::size_t lenOfVirtual = std::string("virtual ").length();
77  if (0 == function.find("virtual ")) {
78  function.erase(0, lenOfVirtual);
79  }
80  // remove leading static
81  std::size_t lenOfStatic = std::string("static ").length();
82  if (0 == function.find("static ")) {
83  function.erase(0, lenOfStatic);
84  }
85  // remove function return declaration for non-constructors
86  size_t openingBraket = function.find('(');
87  size_t space = function.find(' ');
88  if (space < openingBraket) {
89  function.erase(0, space + 1);
90  }
91  // remove function argument declaration
92  openingBraket = function.find('(');
93  size_t closeingBraket = function.rfind(')');
94  if (openingBraket < closeingBraket && std::string::npos != openingBraket) {
95  function.erase(openingBraket + 1, closeingBraket - openingBraket - 1);
96  function.insert(openingBraket + 1, "...");
97  }
98  // handle namespaces in function
99  std::size_t functionNameLength = 47;
100  if (functionNameLength < function.size()) {
101  unsigned int numberOfDoubleColons = 0;
102  size_t nPos = 0;
103  size_t lastDoubleColon = 0;
104  size_t secondLastDoubleColon = 0;
105  while ((nPos = function.find("::", nPos)) != std::string::npos) {
106  numberOfDoubleColons++;
107  secondLastDoubleColon = lastDoubleColon;
108  lastDoubleColon = nPos;
109  nPos += 2;
110  }
111  if (functionNameLength > (function.size() - secondLastDoubleColon + 1)) {
112  function.erase(0, secondLastDoubleColon + 2);
113  function = "..." + function;
114  } else {
115  if (functionNameLength > (function.size() - lastDoubleColon + 1)) {
116  function.erase(0, lastDoubleColon + 2);
117  function = "..." + function;
118  }
119  }
120  // size_t doubleColon = function.find("::");
121  // while (doubleColon < lastDoubleColon) {
122  // function.erase(0, doubleColon + 2);
123  // doubleColon = function.find("::");
124  // }
125  }
126  stream.clear();
127  stream.str("");
128  std::size_t severityLength = 8;
129  std::size_t threadIDLength = 6;
130  std::size_t kindLength = 20;
131  std::size_t lineLength = 6;
132  stream << std::setfill(' ') << timeStr << " " << std::left << std::setw(severityLength) << severity << " "
133  << "[" << std::right << std::setw(threadIDLength) << threadID << "] "
134  << "(" << std::left << std::setw(kindLength) << kind.substr(0, kindLength) << ") "
135  << std::right << std::setw(functionNameLength)
136  << function.substr(0, functionNameLength) << " " << std::right << std::setw(lineLength) << line << ": "
137  << message << std::endl;
138  std::lock_guard<std::mutex> lockGuard(mMutex);
139  if (mFileOpen) {
140  mFileStream << stream.str() << std::flush;
141  } else {
142  mInitialBuffer += stream.str();
143  }
144 }

146 void UKLogger::setLogfileName(const std::string& name) {
147  mFileStream.open(name, std::ios::out);
148  if (mFileStream.good()) {
149  mFileStream.close();
150  UKLOG_WARN("Startup", "Logfile " + name + " exists. Deleting.");
151  if (0 != remove(name.c_str())) {
152  UKLOG_FATAL("Startup", "Could not delete existing logfile " + name + ". Quitting.");
153  // std::cout << mInitialBuffer << std::flush;
154  // std::exit will call the destrector and moveToTerminal()
155  std::exit(EXIT_FAILURE);
156  }
157  }
158  mFileStream.open(name, std::ios::out);
159  if (!mFileStream.is_open()) {
160  UKLOG_FATAL("Startup", "Could not open logfile " + name + ". Quitting.");
161  // std::cout << "UKLogger::setLogfileName should exit" << std::endl << std::flush;
162  // std::cout << mInitialBuffer << std::flush;
163  // std::exit will call the destructor and moveToTerminal()
164  std::exit(EXIT_FAILURE);
165  }
166  std::lock_guard<std::mutex> lockGuard(mMutex);
167  mFileOpen = true;
168  mFileStream << mInitialBuffer;
169 }
